#include "NameWizard.h"
#include "../Game.h"

// Name Wizard
void NameWizard::init() {
    Sprite::init();
}

void NameWizard::update(Timestep ts) {
    Sprite::update(ts);
}

void NameWizard::handleEvent() {
   if (drag(*this, e)) { Wizards::upgradeManager().setSprite(WIZARD); }
}

void NameWizard::render() {
    Rect r = Game::getAbsRect(mRect);
    Game::assets().drawTexture(getImage(), r, NULL);

    Sprite::render();
}
